196def dtm(isostr): 

197 """Get a datetime value from an ISO string representing time.""" 

198 

199 isostr = isostr.rstrip(Z) 

200 try: 

201 date = dt.strptime(isostr, ISO_DTP) 

202 except Exception: 

203 try: 

204 date = dt.strptime(isostr, ISO_DT) 

205 except Exception: 

206 try: 

207 date = dt.strptime(isostr, ISO_D) 

208 except Exception as err: 

209 return (str(err), isostr) 

210 return (E, date)

289def filterModified(modified): 

290 """Filter a provenance trail. 

291 

292 The provenance trail is a list of strings shaped as `"actor on date"` corresponding 

293 to changes in a record. 

294 

295 After filtering we retain for each day only the last modification event per person. 

296 """ 

297 

298 logicM = decomposeM(modified) 

299 chunks = perDay(logicM) 

300 thinned = thinM(chunks) 

301 return composeM(thinned)

331def perDay(modified): 

332 """Auxiliary in provenance filtering: chunk the trails into daily bits.""" 

333 

334 chunks = {} 

335 for m in modified: 

336 chunks.setdefault(dt.date(m[1]), []).append(m) 

337 return [chunks[date] for date in sorted(chunks)]

340def thinM(chunks): 

341 """Auxiliary in provenance filtering: weed out the non-last items per day.""" 

342 

343 modified = [] 

344 nChunks = len(chunks) 

345 for (i, chunk) in enumerate(chunks): 

346 isLast = i == nChunks - 1 

347 people = {} 

348 for m in chunk: 

349 people.setdefault(m[0], []).append(m[1]) 

350 thinned = [] 

351 for (p, dates) in people.items(): 

352 thinned.append((p, sorted(dates)[-1])) 

353 for m in sorted(thinned, key=lambda x: x[1]): 

354 modified.append((m, 2 if isLast else 1)) 

355 return modified
